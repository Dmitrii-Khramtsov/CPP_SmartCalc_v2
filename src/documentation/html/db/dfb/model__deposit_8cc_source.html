38void DepositModel::calculateDeposit(const Input& in, Output& out) {
39 // инициализация векторов изменений
40 ChangeVector additions = in.additions;
41 ChangeVector payments = in.payments;
42
43 // создание объекта данных для расчета
44 CalcData data;
45
46 // инициализация данных для расчета
47 initializeCalculationData(data, in, additions, payments);
48
49 // инициализация выходных данных
50 initializeOutput(out, in);
51
52 // цикл по дням от текущего момента до конца срока вклада
53 for (auto current = in.now; current <= data.end;
54 current += std::chrono::hours(24)) {
55 // обработка текущего дня
56 processDay(current, data, out, in, additions, payments);
57 }
58
59 // обновление начальной записи отчета
60 updateInitialReportRecord(out, in);
61}

102void DepositModel::initializeCalculationData(CalcData& data,
103 const Input& params,
104 ChangeVector& additions,
105 ChangeVector& payments) {
106 // инициализация данных для расчета
107 auto current = params.now;
108 int totalDays = 0;
109
110 // расчет общего количества дней вклада
111 for (int i = 0; i < params.monthsTerm; ++i) {
112 int daysInMonth = getDaysInMonthFromDate(current);
113 totalDays += daysInMonth;
114 // переход к следующему месяцу
115 current += std::chrono::hours(24 * daysInMonth);
116 }
117
118 // установка конечной даты вклада
119 data.end = params.now + std::chrono::hours(24 * totalDays);
120
121 // сортировка записей изменений
122 sortChangeRecords(additions);
123 sortChangeRecords(payments);
124
125 // инициализация вектора дат начисления процентов
126 initPercentsVector(data, params);
127
128 // инициализация начальных значений процентов
129 data.percents = 0;
130 data.percentsByYear = 0;
131}

150void DepositModel::initPercentsVector(CalcData& data, const Input& params) {
151 // инициализация вектора дат начисления процентов
152 data.percentsDate.clear();
153
154 auto current = params.now;
155 while (current < data.end) {
156 switch (params.period) {
157 case Period::EveryDay:
158 data.percentsDate.push_back(current);
159 // переход к следующему дню
160 current += std::chrono::hours(24);
161 break;
162 case Period::EveryMonth:
163 data.percentsDate.push_back(current);
164 // переход к следующему месяцу
165 current += std::chrono::hours(24 * getDaysInMonthFromDate(current));
166 break;
167 case Period::EveryYear:
168 data.percentsDate.push_back(current);
169 // переход к следующему году
170 current += std::chrono::hours(24 * 365);
171 break;
172 }
173 }
174
175 // добавление конечной даты в вектор
176 data.percentsDate.push_back(data.end);
177}

193void DepositModel::processDay(const TimePoint& current, CalcData& data,
194 Output& out, const Input& params,
195 ChangeVector& additions, ChangeVector& payments) {
196 // обработка текущего дня
197 data.havingRecord = false;
198 Report add;
199 add.date = current;
200 add.percents = 0;
201 add.addition = 0;
202 add.remainder = out.remainderDeposit;
203
204 // обработка начисления процентов
205 processInterest(current, add, data, out, params);
206 // обработка пополнений
207 processAdditions(current, add, data, out, additions);
208 // обработка снятий
209 processPayments(current, add, data, out, payments);
210
211 // добавление записи в отчет
212 addToReport(out, add, data.havingRecord);
213
214 // расчет процентов и налогов
215 calculateInterestAndTax(current, data, out, params);
216}

228void DepositModel::processInterest(const TimePoint& current, Report& add,
229 CalcData& data, Output& out,
230 const Input& params) {
231 // обработка начисления процентов
232 if (!data.percentsDate.empty() && current == data.percentsDate.front()) {
233 data.havingRecord = true;
234
235 if (params.capitalisation) {
236 // обновление записи отчета и остатка вклада при капитализации
237 updateReportRecord(add, data.percents, data.havingRecord);
238 out.remainderDeposit += data.percents;
239 }
240
241 // обновление записи отчета
242 add.percents += data.percents;
243 data.percents = 0;
244 data.percentsDate.erase(data.percentsDate.begin());
245 }
246}

291void DepositModel::calculateInterestAndTax(const TimePoint& current,
292 CalcData& data, Output& out,
293 const Input& params) {
294 // расчет процентов за день
295 double interest = depositProfitForDay(out.remainderDeposit, params.rate,
296 daysInYear(current));
297 data.percents += interest;
298 out.totalPercents += interest;
299 data.percentsByYear += interest;
300
301 // расчет налогов, если текущий день - первый день года
302 if (isFirstDayOfYear(current)) {
303 out.totalTax += calculateTax(&data.percentsByYear, params.taxRate,
304 params.maxNonTaxableIncome);
305 }
306}

357void DepositModel::processChangeRecord(const TimePoint& current, Report& add,
358 CalcData& data, Output& out,
359 ChangeVector& changes, bool isAddition) {
360 // обработка записи изменения
361 if (hasChanges(changes, current)) {
362 if (isValidChange(changes.front(), isAddition, out.remainderDeposit)) {
363 // обработка корректной записи изменения
364 processValidChange(add, data, out, changes, isAddition);
365 } else {
366 // обработка некорректной записи изменения
367 processInvalidChange(changes, isAddition);
368 }
369 }

393bool DepositModel::isValidChange(const Change& change, bool isAddition,
394 double remainderDeposit) {
395 // проверка корректности записи изменения
396 return (isAddition && change.sum > 0) ||
397 (!isAddition && remainderDeposit - change.sum >= 0);
398}

410void DepositModel::processValidChange(Report& add, CalcData& data, Output& out,
411 ChangeVector& changes, bool isAddition) {
412 // обработка корректной записи изменения (пополнение или снятие средств)
413 double changeSum = isAddition ? changes.front().sum : -changes.front().sum;
414 updateReportRecord(add, changeSum, data.havingRecord);
415 out.remainderDeposit += changeSum;
416 changes.erase(changes.begin());
417}

429 : "Error: Insufficient funds for payment.")

460double DepositModel::calculateTax(double* sum, double rate,
461 double maxNonTaxableIncome) {
462 // расчет налога
463 double border = maxNonTaxableIncome * rate;
464 double res = 0;
465 if (*sum > border) {
466 res = ((*sum - border) * 13) / 100;
467 }
468 *sum = 0;
469 return res;
470}
