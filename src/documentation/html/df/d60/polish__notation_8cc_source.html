17String ReversePolishNotation::toRPN(const String& infix) {
18 String processed_infix = processedInfix(infix);
19 StringStack operators; // стек для хранения операторов и функций
20 String output; // строка для хранения результата в RPN
21 String token; // временная строка для накопления операндов
22
23 for (size_t i = 0; i < processed_infix.length(); ++i) {
24 switch (isOperandOrOperatorOrFunction(processed_infix, i)) {
25 case OPERAND:
26 handleOperand(output, token, processed_infix, i);
27 break;
28 case OPERATOR:
29 processed_infix[i] == '~'
30 ? handleUnaryMinus(output, token, operators)
31 : handleOperator(output, token, operators, processed_infix[i]);
32 break;
33 case FUNCTION:
34 handleFunction(output, token, operators, processed_infix[i]);
35 break;
36 case BRACKETS:
37 handleParenthesis(output, token, operators, processed_infix[i]);
38 break;
39 }
40 }
41
42 flushToken(output, token);
43 flushOperators(operators, output);
44
45 return trim(output);
46}

313void ReversePolishNotation::handleOperator(String& output,
314 String& token,
315 StringStack& operators,
316 char c) {
317 flushToken(output, token);
318 while (!operators.empty()) {
319 char topOperator = operators.top()[0];
320 int topPriority = getOperatorPriority(topOperator);
321 int currentPriority = getOperatorPriority(c);
322 if (topPriority >= currentPriority) {
323 popAndAppendOperator(operators, output);
324 } else {
325 break;
326 }
327 }
328 operators.push(String(1, c));
329}

355void ReversePolishNotation::handleParenthesis(
356 String& output, String& token, StringStack& operators,
357 char c) {
358 if (c == '(') {
359 operators.push(String(1, c)); // перенести открывающую скобку в стек
360 } else if (c == ')') {
361 flushToken(output, token);
362 while (!operators.empty() && operators.top() != "(") {
363 popAndAppendOperator(operators, output);

378void ReversePolishNotation::processExponent(const String& infix, size_t& i,
379 String& token) {
380 if (i + 1 < infix.size() && (infix[i + 1] == '+' || infix[i + 1] == '-')) {
381 token += infix[++i]; // добавляем '+' или '-' убедимся,
382 // что после '+' или '-' идет число
383 if (i + 1 < infix.size() && std::isdigit(infix[i + 1])) {
384 token += infix[++i];
385 } else {
386 // если после '+' или '-' не следует число, выводим ошибку
387 throw std::invalid_argument(
388 "Invalid format: expected digit after 'e' and sign");
389 }
390 } else if (i + 1 < infix.size() && std::isdigit(infix[i + 1])) {
391 token += infix[++i];
392 } else {
393 // если после 'e' или 'E' не следует число, выводим ошибку
394 throw std::invalid_argument("Invalid format: expected digit after 'e'");
395 }
396}

405void ReversePolishNotation::findScientificNumber(const String& infix,
406 size_t& i,
407 String& token) {
408 bool hasExponent = false;
409 while (i < infix.size() &&
410 (isOperand(infix[i]) || isPartOfExponent(infix[i]))) {
411 token += infix[i];
412 if ((isPartOfExponent(infix[i])) && !hasExponent) {
413 hasExponent = true;
414 processExponent(infix, i, token);
415 }
416 i++;
417 }
418 i--; // компенсируем инкремент i на последнем шаге
419}

428String ReversePolishNotation::trim(const String& str) {
429 size_t first = str.find_first_not_of(' ');
430 if (first == String::npos) return ""; // нет контента
431 size_t last = str.find_last_not_of(' ');
