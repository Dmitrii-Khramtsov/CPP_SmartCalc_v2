56void CreditModel::calculateAnnuity(CrInput in, double &monthly_pay,
57 CrOutput &out) {
58 monthly_pay = round(in.credit * (in.rate / 1200) /
59 (1 - pow(1 + in.rate / 1200, -in.term)) * 100) /
60 100;
61 out.total = monthly_pay * in.term;
62 out.overpayment = out.total - in.credit;
63}

79PaymentVector CreditModel::calculateDifferential(CrInput in, double& first_month_pay, CrOutput &out) {
80 PaymentVector payments;
81 double total_reminder = in.credit;
82 double const_payment = in.credit / in.term;
83 double const_rate = in.rate / 12 / 100; // процентная ставка в месяц
84 double monthly_pay;
85 double interest_pay;
86 out.total = 0;
87
88 int current_day, current_month, current_year;
89 getCurrentDateAndTime(current_day, current_month, current_year);
90
91 // вычисляем платёж в первый месяц
92 first_month_pay = calculateFirstMonthPayment(total_reminder, const_rate, const_payment);
93
94 for (int i = 0; i < in.term; i++) {
95 // вычисляем платеж за месяц
96 calculateMonthlyPayment(total_reminder, const_payment, const_rate, monthly_pay, interest_pay, out);
97
98 // отформатируем дату и добавим детали платежа в вектор платежей
99 std::string formatted_date = formatDate(current_day, current_month, current_year);
100 payments.push_back({monthly_pay, interest_pay, formatted_date, total_reminder, const_payment});
101
102 // увеличение месяца и года
103 incrementMonthAndYear(current_day, current_month, current_year);
104 }
105
106 // расчитаем переплату
107 out.overpayment = out.total - in.credit;
108 return payments;
109}

125void CreditModel::calculateMonthlyPayment(double& total_reminder, double const_payment, double const_rate, double& monthly_pay, double& interest_pay, CrOutput& out) {
126 // Вычисляем проценты на оставшуюся сумму долга
127 interest_pay = total_reminder * const_rate;
128
129 // общий платеж за месяц
130 monthly_pay = const_payment + interest_pay;
131
132 // обновляем оставшуюся сумму долга
133 total_reminder -= const_payment;
134
135 // добавляем общий платеж к общей сумме
136 out.total += monthly_pay;
137}
